42 inline void sendData(hls::stream<ap_uint<8 * PARALLEL_BYTE> >& inStream,
43  ap_uint<8 * PARALLEL_BYTE * 2>& accRegister,
44  uint8_t& bytesInAcc,
45  hls::stream<ap_uint<8 * PARALLEL_BYTE> >& outStream,
46  uint64_t size2Write) {
47 #pragma HLS INLINE
48  const uint16_t c_streamWidth = 8 * PARALLEL_BYTE;
49  const uint16_t c_accRegWidthx3 = c_streamWidth * 3;
50  // transfer data module
51  ap_uint<c_accRegWidthx3> wbuf = accRegister;
52  uint8_t bytesWritten = 0;
53  uint8_t updBInAcc = bytesInAcc;
54 
55  uint8_t bitsInAcc = bytesInAcc * 8;
56 // write block data
57 send_data_main_loop:
58  for (int i = 0; i < size2Write; i += PARALLEL_BYTE) {
59 #pragma HLS PIPELINE II = 1
60  if (i < (const int)(size2Write - bytesInAcc)) {
61  wbuf.range((bitsInAcc + c_streamWidth - 1), bitsInAcc) = inStream.read();
62  updBInAcc += PARALLEL_BYTE;
63  }
64  ap_uint<c_streamWidth> tmpV = wbuf;
65  outStream << tmpV;
66 
67  if (i > (const int)(size2Write - PARALLEL_BYTE)) {
68  bytesWritten = size2Write - i;
69  break;
70  }
71  wbuf >>= c_streamWidth;
72  updBInAcc -= PARALLEL_BYTE;
73  }
74  accRegister = (wbuf >> (bytesWritten * 8));
75  bytesInAcc = updBInAcc - bytesWritten;
76 }

78 template <uint8_t PARALLEL_BYTE>
79 int generateFSETable(hls::stream<uint32_t>& fseTableStream,
80  hls::stream<ap_uint<8 * PARALLEL_BYTE> >& inStream,
81  ap_uint<8 * PARALLEL_BYTE * 2>& fseAcc,
82  uint8_t& bytesInAcc,
83  uint8_t& tableLog,
84  uint16_t maxChar,
85  xfSymbolCompMode_t fseMode,
86  xfSymbolCompMode_t prevFseMode,
87  const int16_t* defDistTable,
88  int16_t* prevDistribution) {
89  const uint16_t c_streamWidth = 8 * PARALLEL_BYTE;
90  const uint16_t c_accRegWidth = c_streamWidth * 2;
91  uint32_t bytesAvailable = bytesInAcc;
92  uint8_t bitsAvailable = bytesInAcc * 8;
93  uint32_t bitsConsumed = 0;
94  int totalBytesConsumed = 0;
95  uint32_t totalBitsConsumed = 0;
96  int remaining = 0;
97  int threshold = 0;
98  bool previous0 = false;
99  uint16_t charnum = 0;
100  int16_t normalizedCounter[64];
101  // initialize normalized counter
102  for (int i = 0; i < 64; ++i) normalizedCounter[i] = 0;
103 
104  if (fseMode == FSE_COMPRESSED_MODE) {
105  // read PARALLEL_BYTE bytes
106  if (bytesAvailable < PARALLEL_BYTE) {
107  fseAcc.range((bitsAvailable + c_streamWidth - 1), bitsAvailable) = inStream.read();
108  bytesAvailable += PARALLEL_BYTE;
109  bitsAvailable += c_streamWidth;
110  }
111  uint8_t accuracyLog = (fseAcc & 0xF) + 5;
112  tableLog = accuracyLog;
113  fseAcc >>= 4;
114  bitsAvailable -= 4;
115  totalBitsConsumed = 4;
116 
117  remaining = (1 << accuracyLog) + 1;
118  threshold = 1 << accuracyLog;
119  accuracyLog += 1;
120  fsegenNormDistTable:
121  while ((remaining > 1) & (charnum <= maxChar)) {
122 #pragma HLS PIPELINE II = 1
123  bitsConsumed = 0;
124  // read PARALLEL_BYTE bytes
125  if (bytesAvailable < PARALLEL_BYTE - 1) {
126  fseAcc.range((bitsAvailable + c_streamWidth - 1), bitsAvailable) = inStream.read();
127  bytesAvailable += PARALLEL_BYTE;
128  bitsAvailable += c_streamWidth;
129  }
130  if (previous0) {
131  unsigned n0 = 0;
132  if ((fseAcc & 0xFFFF) == 0xFFFF) {
133  n0 += 24;
134  bitsConsumed += 16;
135  } else if ((fseAcc & 3) == 3) {
136  n0 += 3;
137  bitsConsumed += 2;
138  } else {
139  n0 += fseAcc & 3;
140  bitsConsumed += 2;
141  previous0 = false;
142  }
143  charnum += n0;
144  } else {
145  int16_t max = (2 * threshold - 1) - remaining;
146  int count;
147  uint8_t c1 = 0;
148  if ((fseAcc & (threshold - 1)) < max) {
149  c1 = 1;
150  count = fseAcc & (threshold - 1);
151  bitsConsumed += accuracyLog - 1;
152  } else {
153  c1 = 2;
154  count = fseAcc & (2 * threshold - 1);
155  if (count >= threshold) count -= max;
156  bitsConsumed += accuracyLog;
157  }
158 
159  --count; /* extra accuracy */
160  remaining -= ((count < 0) ? -count : count); /* -1 means +1 */
161  normalizedCounter[charnum++] = count;
162  previous0 = ((count == 0) ? 1 : 0);
163  genDTableIntLoop:
164  while (remaining < threshold) {
165  --accuracyLog;
166  threshold >>= 1;
167  }
168  }
169  fseAcc >>= bitsConsumed;
170  bitsAvailable -= bitsConsumed;
171  totalBitsConsumed += bitsConsumed;
172 
173  bytesAvailable = bitsAvailable >> 3;
174  }
175  totalBytesConsumed += (totalBitsConsumed + 7) >> 3;
176  bytesInAcc = bytesAvailable;
177  // clear accumulator, so that it is byte aligned
178  fseAcc >>= (bitsAvailable & 7);
179  // copy to prevDistribution table
180  for (int i = 0; i < 64; ++i) prevDistribution[i] = normalizedCounter[i];
181  } else if (fseMode == PREDEFINED_MODE) { /*TODO: use fixed table directly*/
182  for (int i = 0; i < maxChar + 1; ++i) {
183  normalizedCounter[i] = defDistTable[i];
184  prevDistribution[i] = defDistTable[i];
185  }
186  charnum = maxChar + 1;
187  } else if (fseMode == RLE_MODE) {
188  // next state for entire table is 0
189  // accuracy log is 0
190  // read PARALLEL_BYTE bytes
191  if (bytesAvailable < 1) {
192  fseAcc.range((bitsAvailable + c_streamWidth - 1), bitsAvailable) = inStream.read();
193  bytesAvailable += PARALLEL_BYTE;
194  bitsAvailable += c_streamWidth;
195  }
196  uint8_t symbol = (uint8_t)fseAcc;
197  fseAcc >>= 8;
198  bytesInAcc = bytesAvailable - 1;
199 
200  fseTableStream << 1; // tableSize
201  fseTableStream << symbol;
202  tableLog = 0;
203 
204  return 1;
205  } else if (fseMode == REPEAT_MODE) {
206  // check if previous mode was RLE
207  fseTableStream << 0xFFFFFFFF; // all 1's to indicate use previous table
208  return 0;
209  } else {
210  // else -> Error: invalid fse mode
211  return 0;
212  }
213 
214  // Table Generation
215  const uint32_t tableSize = 1 << tableLog;
216  uint32_t highThreshold = tableSize - 1;
217  uint16_t symbolNext[64];
218  int16_t symTable[513];
219  // initialize table
220  for (uint32_t i = 0; i < tableSize; ++i) symTable[i] = 0;
221 
222 fse_gen_next_symbol_table:
223  for (uint16_t i = 0; i < charnum; i++) {
224 #pragma HLS PIPELINE II = 1
225  if (normalizedCounter[i] == -1) {
226  symTable[highThreshold] = i; // symbol, assign lower 8-bits
227  --highThreshold;
228  symbolNext[i] = 1;
229  } else {
230  symbolNext[i] = normalizedCounter[i];
231  }
232  }
233 
234  uint32_t mask = tableSize - 1;
235  const uint32_t step = (tableSize >> 1) + (tableSize >> 3) + 3;
236  uint32_t pos = 0;
237 
238 fse_gen_symbol_table_outer:
239  for (uint16_t i = 0; i < charnum; ++i) {
240  fse_gen_symbol_table:
241  for (int j = 0; j < normalizedCounter[i];) {
242 #pragma HLS PIPELINE II = 1
243  if (pos > highThreshold) {
244  pos = (pos + step) & mask;
245  } else {
246  symTable[pos] = i;
247  pos = (pos + step) & mask;
248  ++j;
249  }
250  }
251  }
252  // FSE table
253  fseTableStream << tableSize;
254 gen_fse_final_table:
255  for (uint32_t i = 0; i < tableSize; i++) {
256 #pragma HLS PIPELINE II = 1
257  uint8_t symbol = (uint8_t)symTable[i];
258  uint16_t nextState = symbolNext[symbol]++;
259  uint16_t nBits = (uint8_t)(tableLog - (31 - __builtin_clz(nextState)));
260  uint32_t tblVal = ((uint32_t)((nextState << nBits) - tableSize) << 16) + ((uint32_t)nBits << 8) + symbol;
261  fseTableStream << tblVal;
262  }
263 
264  return totalBytesConsumed;
265 }

267 template <uint8_t PARALLEL_BYTE, uint8_t BLOCK_SIZE_KB, uint8_t BSWIDTH>
268 void fseStreamStates(uint32_t* litFSETable,
269  uint32_t* oftFSETable,
270  uint32_t* mlnFSETable,
271  ap_uint<BSWIDTH>* bitStream,
272  int byteIndx,
273  uint8_t lastByteValidBits,
274  uint32_t seqCnt,
275  uint8_t* accuracyLog,
276  hls::stream<uint8_t>& symbolStream,
277  hls::stream<ap_uint<32> >& bsWordStream,
278  hls::stream<ap_uint<5> >& extraBitStream) {
279  // fetch fse states from fse sequence bitstream and stream out for further processing
280  const uint16_t c_streamWidth = 8 * PARALLEL_BYTE;
281  const uint16_t c_accRegWidth = c_streamWidth * 2;
282  const uint8_t c_bsBytes = BSWIDTH / 8;
283 
284  uint16_t fseStateLL, fseStateOF, fseStateML; // literal_length, offset, match_length states
285  FseBSState bsStateLL, bsStateOF, bsStateML; // offset, match_length and literal_length
286  ap_uint<c_accRegWidth> acchbs;
287  uint8_t bitsInAcc = lastByteValidBits;
288  uint8_t bitsToRead = 0;
289  uint8_t bytesToRead = 0;
290 
291  acchbs.range(BSWIDTH - 1, 0) = bitStream[byteIndx--];
292  uint8_t byte_0 = acchbs.range(bitsInAcc - 1, bitsInAcc - 8);
293 // find valid last bit, bitstream read in reverse order
294 fsedseq_skip_zero:
295  for (uint8_t i = 7; i >= 0; --i) {
296 #pragma HLS UNROLL
297 #pragma HLS LOOP_TRIPCOUNT min = 1 max = 7
298  if ((byte_0 & (1 << i)) > 0) {
299  --bitsInAcc;
300  break;
301  }
302  --bitsInAcc; // discount higher bits which are zero
303  }
304 
305 fsedseq_fill_acc:
306  while ((bitsInAcc + BSWIDTH < c_accRegWidth) && (byteIndx > -1)) {
307 #pragma HLS PIPELINE II = 1
308  acchbs <<= BSWIDTH;
309  acchbs.range(BSWIDTH - 1, 0) = bitStream[byteIndx--];
310  bitsInAcc += BSWIDTH;
311  }
312  // Read literal_length, offset and match_length states from input stream
313  // get *accuracyLog bits from higher position in accuracyLog, mask out higher scrap bits
314  uint64_t mskLL = ((1 << accuracyLog[0]) - 1);
315  fseStateLL = ((acchbs >> (bitsInAcc - accuracyLog[0])) & mskLL);
316  uint64_t mskOF = ((1 << accuracyLog[1]) - 1);
317  fseStateOF = ((acchbs >> (bitsInAcc - (accuracyLog[0] + accuracyLog[1]))) & mskOF);
318  uint64_t mskML = ((1 << accuracyLog[2]) - 1);
319  fseStateML = ((acchbs >> (bitsInAcc - (accuracyLog[0] + accuracyLog[1] + accuracyLog[2]))) & mskML);
320 
321  bitsInAcc -= (accuracyLog[0] + accuracyLog[1] + accuracyLog[2]);
322 
323  enum FSEDState_t { LITLEN = 0, MATLEN, OFFSET, NEXTSTATE };
324  FSEDState_t smState = OFFSET;
325  uint8_t bitCntLML, bitCntLMO;
326  uint32_t bswLL, bswML, bswOF;
327 
328 decode_sequence_bitStream:
329  while (seqCnt) {
330 #pragma HLS PIPELINE II = 1
331  // read data to bitstream if necessary
332  if (bitsInAcc < c_streamWidth && byteIndx > -1) {
333  auto tmp = bitStream[byteIndx--];
334  acchbs = (acchbs << BSWIDTH) + tmp;
335  bitsInAcc += BSWIDTH;
336  }
337  uint32_t stateVal;
338  uint8_t extraBits;
339  // get state values and stream
340  // stream fse metadata to decoding unit
341  if (smState == LITLEN) {
342  bsWordStream << bswOF;
343 
344  stateVal = litFSETable[fseStateLL]; // literal_length
345  bsStateLL.symbol = stateVal & 0x000000FF;
346  bsStateLL.nextState = (stateVal >> 16) & 0x0000FFFF;
347  bsStateLL.bitCount = (stateVal >> 8) & 0x000000FF;
348 
349  extraBits = c_extraBitsLL[bsStateLL.symbol]; // max 16-bits
350  bitsInAcc -= extraBits;
351  symbolStream << bsStateLL.symbol;
352  bswLL = acchbs >> bitsInAcc;
353  extraBitStream << extraBits;
354 
355  bitCntLML += bsStateLL.bitCount;
356  bitCntLMO += bsStateLL.bitCount;
357  smState = NEXTSTATE;
358  } else if (smState == MATLEN) {
359  stateVal = mlnFSETable[fseStateML]; // match_length
360  bsStateML.symbol = stateVal & 0x000000FF;
361  bsStateML.nextState = (stateVal >> 16) & 0x0000FFFF;
362  bsStateML.bitCount = (stateVal >> 8) & 0x000000FF;
363 
364  extraBits = c_extraBitsML[bsStateML.symbol]; // max 16-bits
365  bitsInAcc -= extraBits;
366  symbolStream << bsStateML.symbol;
367  bswML = acchbs >> bitsInAcc;
368  extraBitStream << extraBits;
369 
370  bitCntLML = bsStateML.bitCount;
371  bitCntLMO += bsStateML.bitCount;
372  smState = LITLEN;
373  } else if (smState == OFFSET) {
374  // get the state codes for offset, match_length and literal_length
375  stateVal = oftFSETable[fseStateOF]; // offset
376  bsStateOF.symbol = stateVal & 0x000000FF;
377  bsStateOF.nextState = (stateVal >> 16) & 0x0000FFFF;
378  bsStateOF.bitCount = (stateVal >> 8) & 0x000000FF;
379 
380  extraBits = bsStateOF.symbol; // also represents extra bits to be read, max 31-bits
381  bitsInAcc -= extraBits;
382  symbolStream << bsStateOF.symbol;
383  bswOF = acchbs >> bitsInAcc;
384 
385  bsWordStream << bswLL;
386  bitCntLMO = bsStateOF.bitCount;
387  smState = MATLEN;
388  } else {
389  bsWordStream << bswML;
390  // update state for next sequence
391  // read bits to get states for literal_length, match_length, offset
392  // accumulator must contain these many bits can be max 26-bits
393  mskLL = (((uint64_t)1 << bsStateLL.bitCount) - 1);
394  fseStateLL = ((acchbs >> (bitsInAcc - bsStateLL.bitCount)) & mskLL);
395  fseStateLL += bsStateLL.nextState;
396 
397  mskML = (((uint64_t)1 << bsStateML.bitCount) - 1);
398  fseStateML = ((acchbs >> (bitsInAcc - bitCntLML)) & mskML);
399  fseStateML += bsStateML.nextState;
400 
401  mskOF = (((uint64_t)1 << bsStateOF.bitCount) - 1);
402  fseStateOF = ((acchbs >> (bitsInAcc - bitCntLMO)) & mskOF);
403  fseStateOF += bsStateOF.nextState;
404 
405  bitsInAcc -= bitCntLMO;
406 
407  --seqCnt;
408  smState = OFFSET;
409  }
410  }
411  bsWordStream << bswLL;
412 }

414 template <int LMO_WIDTH>
415 void fseDecodeStates(hls::stream<uint8_t>& symbolStream,
416  hls::stream<ap_uint<32> >& bsWordStream,
417  hls::stream<ap_uint<5> >& extraBitStream,
418  uint32_t seqCnt,
419  uint32_t litCnt,
420  ap_uint<LMO_WIDTH>* prevOffsets,
421  hls::stream<ap_uint<LMO_WIDTH> >& litLenStream,
422  hls::stream<ap_uint<LMO_WIDTH> >& offsetStream,
423  hls::stream<ap_uint<LMO_WIDTH> >& matLenStream,
424  hls::stream<bool>& litlenValidStream) {
425  // calculate literal length, match length and offset values, stream them for sequence execution
426  enum FSEDecode_t { LITLEN = 0, MATLEN, OFFSET_CALC, OFFSET_WNU };
427  FSEDecode_t sqdState = OFFSET_CALC;
428  ap_uint<LMO_WIDTH> offsetVal;
429  ap_uint<LMO_WIDTH> litLenCode;
430  uint8_t ofi;
431  bool checkLL = false;
432  bsWordStream.read(); // dump first word
433 decode_sequence_codes:
434  while (seqCnt) {
435 #pragma HLS PIPELINE II = 1
436  if (sqdState == OFFSET_CALC) {
437  // calculate offset and set prev offsets
438  auto symbol = symbolStream.read();
439  auto bsWord = bsWordStream.read();
440  auto extBit = symbol;
441  uint32_t extVal = bsWord & (((uint64_t)1 << extBit) - 1);
442  offsetVal = (1 << symbol) + extVal;
443 
444  ofi = 3;
445  if (offsetVal > 3) {
446  offsetVal -= 3;
447  checkLL = false;
448  } else {
449  checkLL = true;
450  }
451  sqdState = MATLEN;
452  } else if (sqdState == MATLEN) {
453  // calculate match length
454  auto symbol = symbolStream.read();
455  auto bsWord = bsWordStream.read();
456  auto extBit = extraBitStream.read();
457  uint16_t extVal = (bsWord & (((uint64_t)1 << extBit) - 1));
458  ap_uint<LMO_WIDTH> matchLenCode = c_baseML[symbol] + extVal;
459  matLenStream << matchLenCode;
460 
461  sqdState = LITLEN;
462  } else if (sqdState == LITLEN) {
463  // calculate literal length
464  auto symbol = symbolStream.read();
465  auto bsWord = bsWordStream.read();
466  auto extBit = extraBitStream.read();
467  uint16_t extVal = (bsWord & (((uint64_t)1 << extBit) - 1));
468  litLenCode = c_baseLL[symbol] + extVal;
469  litlenValidStream << 1;
470  litLenStream << litLenCode;
471  litCnt -= litLenCode;
472 
473  // update offset as per literal length
474  if (checkLL) {
475  // repeat offsets 1 - 3
476  if (litLenCode == 0) {
477  if (offsetVal == 3) {
478  offsetVal = prevOffsets[0] - 1;
479  ofi = 2;
480  } else {
481  ofi = offsetVal;
482  offsetVal = prevOffsets[offsetVal];
483  }
484  } else {
485  ofi = offsetVal - 1;
486  offsetVal = prevOffsets[offsetVal - 1];
487  }
488  }
489  checkLL = false;
490  sqdState = OFFSET_WNU;
491  } else {
492  // OFFSET_WNU: write offset and update previous offsets
493  offsetStream << offsetVal;
494  // shift previous offsets
495  auto prev1 = prevOffsets[1];
496  if (ofi > 1) {
497  prevOffsets[2] = prev1;
498  }
499  if (ofi > 0) {
500  prevOffsets[1] = prevOffsets[0];
501  prevOffsets[0] = offsetVal;
502  }
503 
504  sqdState = OFFSET_CALC;
505  --seqCnt;
506  }
507  }
508  if (litCnt > 0) {
509  litlenValidStream << 1;
510  litLenStream << litCnt;
511  matLenStream << 0;
512  offsetStream << 0;
513  }
514 }

552 template <uint8_t PARALLEL_BYTE, uint8_t BLOCK_SIZE_KB, uint8_t LMO_WIDTH>
553 inline void fseDecode(ap_uint<64> accword,
554  uint8_t bytesInAcc,
555  hls::stream<ap_uint<8 * PARALLEL_BYTE> >& inStream,
556  uint32_t* litFSETable,
557  uint32_t* oftFSETable,
558  uint32_t* mlnFSETable,
559  uint32_t seqCnt,
560  uint32_t litCount,
561  uint32_t remBlockSize,
562  uint8_t* accuracyLog,
563  ap_uint<LMO_WIDTH>* prevOffsets,
564  hls::stream<ap_uint<LMO_WIDTH> >& litLenStream,
565  hls::stream<ap_uint<LMO_WIDTH> >& offsetStream,
566  hls::stream<ap_uint<LMO_WIDTH> >& matLenStream,
567  hls::stream<bool>& litlenValidStream) {
568  // decode fse encoded bitstream using prebuilt fse tables
569  const uint16_t c_streamWidth = 8 * PARALLEL_BYTE;
570  const uint16_t c_accRegWidth = c_streamWidth * 2;
571  const uint16_t c_BSWidth = c_streamWidth;
572  const uint8_t c_bsBytes = c_BSWidth / 8;
573 
574  ap_uint<c_BSWidth> bitStream[(BLOCK_SIZE_KB * 1024) / c_bsBytes];
575 #pragma HLS BIND_STORAGE variable = bitStream type = ram_t2p impl = bram
576 
577  uint8_t bitsInAcc = bytesInAcc * 8;
578 
579  // copy data from bitstream to buffer

612 template <uint8_t PARALLEL_BYTE>
613 void fseDecodeHuffWeight(hls::stream<ap_uint<8 * PARALLEL_BYTE> >& inStream,
614  uint32_t remSize,
615  ap_uint<8 * PARALLEL_BYTE * 2>& accHuff,
616  uint8_t& bytesInAcc,
617  uint8_t accuracyLog,
618  uint32_t* fseTable,
619  uint8_t* weights,
620  uint16_t& weightCnt,
621  uint8_t& huffDecoderTableLog) {
622  //#pragma HLS INLINE
623  const uint16_t c_streamWidth = 8 * PARALLEL_BYTE;
624  uint8_t bitsInAcc = bytesInAcc * 8;
625 
626  uint8_t bitStream[128];
627  //#pragma HLS BIND_STORAGE variable = bitStream type = ram_t2p impl = bram
628  int bsByteIndx = remSize - 1;
629 
630  // copy data from bitstream to buffer
631  uint32_t itrCnt = 1 + ((remSize - 1) / PARALLEL_BYTE);
632  uint32_t k = 0;
633 fseDecHF_read_input:
634  for (uint32_t i = 0; i < itrCnt; ++i) {
635  accHuff.range(bitsInAcc + c_streamWidth - 1, bitsInAcc) = inStream.read();
636  bitsInAcc += c_streamWidth;
637  for (uint8_t j = 0; j < PARALLEL_BYTE && k < remSize; ++j, ++k) {
638 #pragma HLS PIPELINE II = 1
639  bitStream[k] = accHuff.range(7, 0);
640  accHuff >>= 8;
641  bitsInAcc -= 8;
642  }
643  }
644  bytesInAcc = bitsInAcc >> 3;
645 
646  // decode FSE bitStream using fse table to get huffman weights
647  // skip initial 0 bits and single 1 bit
648  uint32_t accState;
649  uint32_t codeIdx = 0;
650  uint8_t bitsToRead = 0;
651  bitsInAcc = 0;
652  int32_t rembits = remSize * 8;
653 
654  uint8_t fseState[2];
655  FseBSState bsState[2];
656  // find beginning of the stream
657  accState = bitStream[bsByteIndx--];
658  bitsInAcc = 8;
659  for (uint8_t i = 7; i >= 0; --i) {
660 #pragma HLS UNROLL
661  if ((accState & (1 << i)) > 0) {
662  --bitsInAcc;
663  break;
664  }
665  --bitsInAcc; // discount higher bits which are zero
666  }
667  rembits -= (8 - bitsInAcc);
668 
669  // Read bits needed for first two states
670  bitsToRead = accuracyLog * 2;
671  if (bitsToRead > bitsInAcc) {
672  uint8_t bytesToRead = 1 + ((bitsToRead - bitsInAcc - 1) / 8);
673  for (uint8_t i = 0; i < bytesToRead; ++i) {
674  uint8_t tmp = bitStream[bsByteIndx--];
675  accState <<= 8;
676  accState += tmp;
677  bitsInAcc += 8;
678  }
679  }
680  // Read initial state1 and state2
681  // get readBits bits from higher position in accuracyLog, mask out higher scrap bits
682  // read state 1
683  bitsInAcc -= accuracyLog;
684  uint64_t msk = ((1 << accuracyLog) - 1);
685  fseState[0] = ((accState >> bitsInAcc) & msk);
686  // read state 2
687  bitsInAcc -= accuracyLog;
688  msk = ((1 << accuracyLog) - 1);
689  fseState[1] = ((accState >> bitsInAcc) & msk);
690  rembits -= (accuracyLog * 2);
691 
692  bool stateIdx = 0; // 0 for even, 1 for odd
693  bool overflow = false;
694  uint32_t totalWeights = 0;
695 fse_decode_huff_weights:
696  while (1) {
697 #pragma HLS PIPELINE II = 1
698  // get the weight, bitCount and nextState
699  uint32_t stateVal = fseTable[fseState[stateIdx]];
700  uint8_t cw = (uint8_t)(stateVal & 0xFF);
701  weights[codeIdx++] = cw;
702  totalWeights += (1 << cw) >> 1;
703 
704  // overflow
705  if (rembits < 0) break;
706 
707  // get other values
708  bsState[stateIdx].nextState = (stateVal >> 16) & 0x0000FFFF;
709  bsState[stateIdx].bitCount = (stateVal >> 8) & 0x000000FF;
710  uint8_t bitsToRead = bsState[stateIdx].bitCount;
711  if (bitsToRead > bitsInAcc) {
712  uint8_t tmp = 0;
713  if (bsByteIndx > -1) {
714  // max 1 read is required, since accuracy log <= 6
715  tmp = bitStream[bsByteIndx--];
716  }
717  accState <<= 8;
718  accState += tmp;
719  bitsInAcc += 8;
720  }
721  // get next fse state
722  bitsInAcc -= bitsToRead;
723  uint8_t msk = ((1 << bitsToRead) - 1);
724  fseState[stateIdx] = ((accState >> bitsInAcc) & msk);
725  fseState[stateIdx] += bsState[stateIdx].nextState;
726  rembits -= bitsToRead;
727 
728  // switch state flow
729  stateIdx = (stateIdx + 1) & 1; // 0 if 1, 1 if 0
730  }
731  huffDecoderTableLog = 1 + (31 - __builtin_clz(totalWeights));
732  // add last weight
733  uint16_t lw = (1 << huffDecoderTableLog) - totalWeights;
734  weights[codeIdx++] = 1 + (31 - __builtin_clz(lw));
735  weightCnt = codeIdx;
736 }

738 template <int MAX_CODELEN>
739 void huffGenLookupTable(uint8_t* weights, HuffmanTable* huffTable, uint8_t accuracyLog, uint16_t weightCnt) {
740  // create huffman lookup table
741  // regenerate huffman tree using literal bit-lengths
742  typedef ap_uint<MAX_CODELEN + 1> LCL_Code_t;
743  LCL_Code_t first_codeword[MAX_CODELEN + 1];
744  ap_uint<32> bitlen_histogram[MAX_CODELEN + 1];
745  ap_uint<4> bitlens[256];
746 #pragma HLS ARRAY_PARTITION variable = first_codeword complete
747 #pragma HLS ARRAY_PARTITION variable = bitlen_histogram complete
748 
749  uint16_t codes[256];
750 // initialize first_codeword and bitlength histogram
751 hflkpt_init_blen_hist:
752  for (uint8_t i = 0; i < MAX_CODELEN + 1; ++i) {
753 #pragma HLS PIPELINE II = 1
754 #pragma HLS LOOP_TRIPCOUNT min = MAX_CODELEN max = MAX_CODELEN
755  bitlen_histogram[i] = 0;
756  }
757 // read bit-lengths
758 hflkpt_fill_blen_hist:
759  for (uint16_t i = 0; i < weightCnt; ++i) {
760 #pragma HLS PIPELINE II = 1
761  // convert weight to bitlen
762  uint8_t cblen = weights[i];
763  bitlen_histogram[cblen]++;
764  if (cblen > 0) cblen = (accuracyLog + 1 - cblen);
765  bitlens[i] = cblen;
766  }
767 
768  // generate first codes
769  first_codeword[0] = bitlen_histogram[0];
770 
771  uint16_t nextCode = 0;
772 hflkpt_initial_codegen:
773  for (uint8_t i = 1; i < accuracyLog + 1; ++i) {
774 #pragma HLS PIPELINE II = 1
775 #pragma HLS LOOP_TRIPCOUNT min = 0 max = 8
776  uint16_t cur = nextCode;
777  nextCode += (bitlen_histogram[i] << (i - 1));
778  first_codeword[i] = cur;
779  }
780 
781 hflkpt_codegen_outer:
782  for (int i = 0; i < weightCnt; ++i) {
783  uint32_t hfw = weights[i];
784  const uint32_t len = (1 << hfw) >> 1;
785  const auto fcw = first_codeword[hfw];
786  hflkpt_codegen:
787  for (uint16_t u = fcw; u < fcw + len; ++u) {
788 #pragma HLS PIPELINE II = 1
789  huffTable[u].symbol = i;
790  huffTable[u].bitlen = bitlens[i];
791  }
792  first_codeword[hfw] = fcw + len;
793  }
794 }

1038  continue;
1039  }

1041  bitsInAcc -= bitsWritten;
1042  bytesInAcc = bitsInAcc >> 3;

1046  }

1049  uint32_t tmp = bitStream[rIdx];
1050  huffBitStream << tmp;

1057  if (wsi < streamCnt) fetchMode = 1;
1058  // either previous streamMode ended quicker than next fetchMode or streamCnt reached
1059  if (wsi == rsi) streamMode = 0;
1060  } else {
1061  rIdx -= smInc;
1062  }
1063  }
1064  // end condition
1065  if (!(fetchMode | streamMode)) done = 1;
1066  }
