39 template <int PARALLEL_BYTES>
40 void lzLiteralUpsizer(hls::stream<ap_uint<8> >& inStream,
41  hls::stream<uint8_t>& inStreamSize,
42  hls::stream<ap_uint<PARALLEL_BYTES * 8> >& litStream) {
43  const uint8_t c_parallelBit = PARALLEL_BYTES * 8;
44  const uint8_t c_maxLitLen = 128;
45  ap_uint<c_parallelBit> outBuffer;
46  uint8_t idx = 0;
47 
48  uint8_t i = 0;
49  bool outFlag = false;
50 lzLiteralUpsizer:
51  for (uint8_t size = inStreamSize.read(); size != (c_maxLitLen + 1);) {
52 #pragma HLS PIPELINE II = 1
53  if (size) {
54  outBuffer.range((idx + 1) * 8 - 1, idx * 8) = inStream.read();
55  if (i == (size - 1)) {
56  size = inStreamSize.read();
57  outFlag = true;
58  idx = 0;
59  i = 0;
60  } else {
61  i++;
62  if (idx == (PARALLEL_BYTES - 1)) {
63  idx = 0;
64  outFlag = true;
65  } else {
66  idx++;
67  outFlag = false;
68  }
69  }
70  if (outFlag) litStream << outBuffer;
71  } else {
72  size = inStreamSize.read();
73  }
74  }
75 }

77 template <class SIZE_DT = uint8_t>
78 void lzProcessingUnit(hls::stream<ap_uint<32> >& inStream,
79  hls::stream<bool>& inEndOfStream,
80  hls::stream<SIZE_DT>& litLenStream,
81  hls::stream<uint8_t>& lenStream,
82  hls::stream<SIZE_DT>& matchLenStream,
83  hls::stream<ap_uint<16> >& offsetStream,
84  hls::stream<ap_uint<8> >& outStream) {
85  ap_uint<32> inValue, nextValue;
86  const int c_maxLitLen = 128;
87  uint16_t offset = 0;
88  uint16_t matchLen = 0;
89  uint8_t litLen = 0;
90  uint8_t outLitLen = 0;
91 
92  bool eosFlag = inEndOfStream.read();
93 
94  nextValue = inStream.read();
95  for (; eosFlag == false;) {
96 #pragma HLS PIPELINE II = 1
97  inValue = nextValue;
98  nextValue = inStream.read();
99  eosFlag = inEndOfStream.read();
100 
101  offset = inValue.range(15, 0);
102  matchLen = inValue.range(31, 16);
103 
104  bool outFlag, outStreamFlag;
105  if (matchLen) {
106  outFlag = true;
107  outLitLen = litLen;
108  litLen = 0;
109  outStreamFlag = false;
110  } else {
111  outStreamFlag = true;
112  outLitLen = litLen + 1;
113  if (litLen == c_maxLitLen - 1) {
114  outFlag = true;
115  litLen = 0;
116  } else {
117  outFlag = false;
118  litLen++;
119  }
120  }
121  if (outStreamFlag) {
122  outStream << inValue.range(7, 0);
123  }
124  if (outFlag) {
125  litLenStream << outLitLen;
126  offsetStream << offset;
127  matchLenStream << matchLen;
128  lenStream << outLitLen;
129  }
130  }
131 
132  if (litLen) {
133  litLenStream << litLen;
134  lenStream << litLen;
135  offsetStream << 0;
136  matchLenStream << 0;
137  }
138 
139  lenStream << c_maxLitLen + 1;
140  litLenStream << 0;
141  offsetStream << 0;
142  matchLenStream << 0;
143 }

191 #pragma HLS PIPELINE II = 1

193  outKStream << t1;
