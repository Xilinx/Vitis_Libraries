207 template <typename T>
208 struct zlib_aligned_allocator {
209  using value_type = T;
210  T* allocate(std::size_t num) {
211  void* ptr = nullptr;
212  if (posix_memalign(&ptr, page_aligned_mem, num * sizeof(T))) throw std::bad_alloc();
213 
214  // madvise is a system call to allocate
215  // huge pages. By allocating huge pages
216  // for memory allocation improves overall
217  // time by reduction in page initialization
218  // in next step.
219  madvise(ptr, num, MADV_HUGEPAGE);
220 
221  T* array = reinterpret_cast<T*>(ptr);
222 
223  // Write a value in each virtual memory page to force the
224  // materialization in physical memory to avoid paying this price later
225  // during the first use of the memory
226  for (std::size_t i = 0; i < num; i += (page_aligned_mem)) array[i] = 0;
227  return array;
228  }
