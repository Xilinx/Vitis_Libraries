402 
403  if (incr != 0) {
404  huff &= incr - 1;

406  } else

408 
409  sym++;
410

412  if (len == max) break;
413  len = lens[codeBuffer[sym]];
414  }
415 
416  if (len > root && (huff & mask) != low) {
417  if (drop == 0) {
418  drop = root;
419  min = 0;
420  is_extra = true;
421  }
422 
423  extra_idx += min;
424  nptr_extra += min;

426  left = (int)(1 << curr);
427

431 #pragma HLS PIPELINE II = 1
432 #pragma HLS UNROLL FACTOR = 2

436  sum++;
437  }

440  table[low] = ((uint32_t)curr << 24) | ((uint32_t)root << 16) | extra_idx;
441  }
442  }
