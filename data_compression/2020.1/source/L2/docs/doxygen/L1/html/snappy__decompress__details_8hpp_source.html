41 static void snappyHeaderProcessing(hls::stream<ap_uint<PARALLEL_BYTES * 8> >& inStream,
42  hls::stream<ap_uint<PARALLEL_BYTES * 8> >& outStream,
43  hls::stream<dt_blockInfo>& blockInfoStream,
44  const uint32_t inputSize) {
45  // Snappy Header states
46  enum snappyDecompressHeaderStates { READ_COMP_LEN, WRITE_DATA };

58  // process first 10 bytes
59  ap_uint<c_parallelBit> temp1 = inStream.read();

62

67  }
68

70 #pragma HLS PIPELINE II = 1
71  inputWindow.range((i + 1) * c_parallelBit - 1, i * c_parallelBit) = inStream.read();
72  readBytes += PARALLEL_BYTES;
73  }
74 
75  inputIdx += 2;
76  dt_blockInfo blockInfo;
77  uint8_t incrInputIdx = 0;
78 
79  while ((processedBytes + inputIdx) < inputSize) {
80  if (inputIdx >= PARALLEL_BYTES) {
81  inputWindow >>= c_parallelBit;
82  inputIdx = inputIdx - PARALLEL_BYTES;
83  processedBytes += PARALLEL_BYTES;
84  if (readBytes < inputSize) {
85  ap_uint<c_parallelBit> input = inStream.read();
86  inputWindow.range(8 * c_parallelBit - 1, 7 * c_parallelBit) = input;
87  readBytes += PARALLEL_BYTES;
88  }
89  }
90 
91  SIZE_DT chunkSize = 0;
92 
93  ap_uint<32> inValue = inputWindow >> (inputIdx * 8);
94  SIZE_DT temp = inValue.range(31, 24);
95  temp <<= 16;
96  chunkSize |= temp;
97  temp = 0;
98  temp = inValue.range(23, 16);
99  temp <<= 8;
100  chunkSize |= temp;
101  temp = 0;
102  chunkSize |= inValue.range(15, 8);
103 
104  origCompLen = chunkSize - 4;
105  compLen = origCompLen;
106 
107  // 4 bytes processed and remaining 4 bytes skipped
108  incrInputIdx = 8;
109 
110  uint8_t chunkIdx = inValue.range(7, 0);
111 
112  if (chunkIdx == 0x01) {
113  blockInfo.storedBlock = 1;
114  } else {
115  blockInfo.storedBlock = 0;
116  uint8_t blkSizeProcessedBytes = 0;
117  ap_uint<16> value = inputWindow >> ((inputIdx + incrInputIdx) * 8);
118 
119  bool c0 = ((value.range(7, 7)) == 1);
120  bool c1 = ((value.range(15, 15)) == 1);
121  if (c0 & c1) {
122  incrInputIdx = 11;
123  blkSizeProcessedBytes = 3;
124  } else if (c0) {
125  incrInputIdx = 10;
126  blkSizeProcessedBytes = 2;
127  } else {
128  incrInputIdx = 9;
129  blkSizeProcessedBytes = 1;
130  }
131  origCompLen = origCompLen - blkSizeProcessedBytes;
132  compLen = origCompLen;
133  }
134 
135  inputIdx += incrInputIdx;
136  blockInfo.compressSize = origCompLen;
137  // write blockInfo to stream
138  blockInfoStream << blockInfo;
139 
140  SIZE_DT len = compLen;
141  for (uint32_t blockData = 0; blockData < compLen; blockData += PARALLEL_BYTES) {
142 #pragma HLS PIPELINE II = 1
143  if (inputIdx >= PARALLEL_BYTES) {
144  inputWindow >>= c_parallelBit;
145  inputIdx = inputIdx - PARALLEL_BYTES;
146  processedBytes += PARALLEL_BYTES;
147  if (readBytes < inputSize) {
148  ap_uint<c_parallelBit> input = inStream.read();
149  inputWindow.range(8 * c_parallelBit - 1, 7 * c_parallelBit) = input;
150  readBytes += PARALLEL_BYTES;
151  }
152  }
153 
154  outStreamValue = inputWindow >> (inputIdx * 8);
155  outStream << outStreamValue;
156 
157  if (len >= PARALLEL_BYTES) {
158  inputIdx += PARALLEL_BYTES;
159  len -= PARALLEL_BYTES;
160  } else {
161  inputIdx += len;
162  len = 0;
163  }
164  }
165  }
166  blockInfo.compressSize = 0;
167  // writing 0 to indicate end of data
168  blockInfoStream << blockInfo;
169 }

314 template <int NUM_BLOCKS, int PARALLEL_BYTES, int BLOCK_SIZE = 64, class SIZE_DT = ap_uint<17> >
315 void lzMultiBlockPacker(hls::stream<ap_uint<PARALLEL_BYTES * 8> > lzDataStream[NUM_BLOCKS],
316  hls::stream<bool> lzStreamEos[NUM_BLOCKS],
317  hls::stream<uint32_t> lzSizeStream[NUM_BLOCKS],
318  hls::stream<ap_uint<PARALLEL_BYTES * 8> >& outStream,
319  hls::stream<bool>& outStreamEos,
320  hls::stream<uint32_t>& outSizeStream) {
321  uint32_t outSize = 0;
322  uint32_t blockSizeInBytes = BLOCK_SIZE * 1024;
323  ap_uint<NUM_BLOCKS> is_pending;
324  bool done = false;
325 
326  for (uint8_t i = 0; i < NUM_BLOCKS; i++) {
327 #pragma HLS UNROLL
328  is_pending.range(i, i) = 1;
329  }
330 
331  while (is_pending) {
332  for (int i = 0; i < NUM_BLOCKS; i++) {
333  for (SIZE_DT read = 0; ((read < blockSizeInBytes) && (is_pending.range(i, i) == true));
334  read += PARALLEL_BYTES) {
335 #pragma HLS PIPELINE II = 1
336  ap_uint<PARALLEL_BYTES* 8> outData = lzDataStream[i].read();
337  bool eosFlag = lzStreamEos[i].read();
338  if (eosFlag == false) {
339  outStream << outData;
340  outStreamEos << 0;
341  } else {
342  is_pending.range(i, i) = 0;
343  }
344  }
345  }
346  }
347 
348  outStream << 0;
349  outStreamEos << 1;
350 
351  for (int i = 0; i < NUM_BLOCKS; i++) {
352 #pragma HLS PIPELINE II = 1
353  outSize += lzSizeStream[i].read();
354  }
355  outSizeStream << outSize;
356 }
