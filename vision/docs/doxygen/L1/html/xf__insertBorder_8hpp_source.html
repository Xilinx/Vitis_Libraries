147  else

172  }

177  out_write_en = pad_col_count_left < NPC && input_col_index < (imgInput_ncpr + 1) &&
178  pad_row_count_top <= 0 && (row_index - dy) < in_height;
179  else

192  else

194  } else {

219  pad_col_count_left -= NPC;

222  pad_row_count_top--;
