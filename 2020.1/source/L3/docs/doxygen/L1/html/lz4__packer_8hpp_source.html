43 template <int DATAWIDTH, int BURST_SIZE>
44 void mm2slz4Packer(const ap_uint<DATAWIDTH>* in,
45  ap_uint<DATAWIDTH>* orig_input_data,
46  hls::stream<ap_uint<DATAWIDTH> >& outStream,
47  hls::stream<uint32_t>& outStreamSize,
48  uint32_t* compressd_size,
49  uint32_t* in_block_size,
50  uint32_t no_blocks,
51  uint32_t block_size_in_kb,
52  uint32_t xxhashVal,
53  uint32_t input_size) {
54  const int c_byte_size = 8;
55  const int c_word_size = DATAWIDTH / c_byte_size;
56 
57  ap_uint<DATAWIDTH> headerData = 0;
58 
59  // LZ4 Header Data
60  headerData(7, 0) = xxhashVal;
61  headerData = headerData << 64;
62  headerData(63, 0) = input_size;
63  headerData = headerData << 8;
64  headerData(7, 0) = block_size_in_kb;
65  headerData = headerData << 8;
66  headerData(7, 0) = xf::compression::FLG_BYTE;
67  headerData = headerData << 8;
68  headerData(7, 0) = xf::compression::MAGIC_BYTE_4;
69  headerData = headerData << 8;
70  headerData(7, 0) = xf::compression::MAGIC_BYTE_3;
71  headerData = headerData << 8;
72  headerData(7, 0) = xf::compression::MAGIC_BYTE_2;
73  headerData = headerData << 8;
74  headerData(7, 0) = xf::compression::MAGIC_BYTE_1;
75 
76  // Handle header or residue here
77  uint32_t block_stride = block_size_in_kb * 1024 / 64;
78 
79  uint32_t blkCompSize = 0;
80  uint32_t origSize = 0;
81  uint32_t sizeInWord = 0;
82  uint32_t cBlen = 4;
83  uint32_t hSize = 15;
84  outStreamSize << hSize;
85  outStream << headerData;
86 
87  // Run over number of blocks
88  for (int bIdx = 0; bIdx < no_blocks; bIdx++) {
89  blkCompSize = compressd_size[bIdx];
90  origSize = in_block_size[bIdx];
91  // Put compress block & input block
92  // into streams for next block
93  sizeInWord = (blkCompSize - 1) / c_word_size + 1;
94 
95  // Send size in bytes
96  outStreamSize << cBlen;
97  outStream << blkCompSize;
98  outStreamSize << blkCompSize;
99 
100  // Copy data from global memory to local
101  // Put it into stream
102  for (uint32_t i = 0; i < sizeInWord; i += BURST_SIZE) {
103  uint32_t chunk_size = BURST_SIZE;
104 
105  if (i + BURST_SIZE > sizeInWord) chunk_size = sizeInWord - i;
106 
107  if (blkCompSize == origSize) {
108  memrd1:
109  for (uint32_t j = 0; j < chunk_size; j++) {
110 #pragma HLS PIPELINE II = 1
111  outStream << orig_input_data[block_stride * bIdx + i + j];
112  }
113  } else {
114  memrd2:
115  for (uint32_t j = 0; j < chunk_size; j++) {
116 #pragma HLS PIPELINE II = 1
117  outStream << in[block_stride * bIdx + i + j];
118  }
119  }
120  }
121  }
122  outStreamSize << 0;
123 }
