58 template <int NUM_BLOCKS, int BLOCK_SIZE, int DATAWIDTH, int BURST_SIZE>
59 void mm2multStreamMM1(const ap_uint<DATAWIDTH>* in,
60  hls::stream<ap_uint<DATAWIDTH> > outStream[NUM_BLOCKS],
61  hls::stream<uint16_t> outSizeStream[NUM_BLOCKS],
62  hls::stream<uint32_t> outLz77SizeStream[NUM_BLOCKS],
63  const uint32_t input_idx[NUM_BLOCKS],
64  const uint32_t _input_size[NUM_BLOCKS]) {
65  const int c_byteSize = 8;
66  const int c_wordSize = DATAWIDTH / c_byteSize;
67 
68  ap_uint<NUM_BLOCKS> is_pending;
69  uint32_t read_idx[NUM_BLOCKS];
70  uint32_t read_size[NUM_BLOCKS];
71  uint32_t input_size[NUM_BLOCKS];
72 #pragma HLS ARRAY_PARTITION variable = read_idx dim = 0 complete
73 #pragma HLS ARRAY_PARTITION variable = read_size dim = 0 complete
74 #pragma HLS ARRAY_PARTITION variable = input_size dim = 0 complete
75 
76  for (uint8_t vid = 0; vid < NUM_BLOCKS; vid++) {
77 #pragma HLS UNROLL
78  read_idx[vid] = input_idx[vid];
79  input_size[vid] = _input_size[vid];
80  read_size[vid] = 0;
81  is_pending.range(vid, vid) = 1;
82  }
83 
84  while (is_pending) {
85  parallel_ops:
86  for (uint32_t vid = 0; vid < NUM_BLOCKS; vid++) {
87  bool isFull = (outSizeStream[vid]).full();
88  uint32_t pendingBytes = 0, sizeWrite = 0;
89  uint32_t blkMod = read_size[vid] % BLOCK_SIZE;
90  if (!isFull) {
91  pendingBytes = (input_size[vid] > read_size[vid]) ? (input_size[vid] - read_size[vid]) : 0;
92  is_pending.range(vid, vid) = (pendingBytes > 0) ? 1 : 0;
93  }
94  if (pendingBytes && blkMod == 0) {
95  uint32_t sendingBytes = BLOCK_SIZE;
96  if (pendingBytes < sendingBytes) sendingBytes = pendingBytes;
97  outLz77SizeStream[vid] << sendingBytes;
98  }
99  if (pendingBytes && !isFull) {
100  uint32_t pendingWords = (pendingBytes - 1) / c_wordSize + 1;
101  uint32_t burstSize = (pendingWords > BURST_SIZE) ? BURST_SIZE : pendingWords;
102  sizeWrite = burstSize * c_wordSize;
103  if (read_size[vid] + sizeWrite <= input_size[vid]) {
104  read_size[vid] += sizeWrite;
105  } else {
106  sizeWrite = input_size[vid] - read_size[vid];
107  read_size[vid] = input_size[vid];
108  }
109 
110  outSizeStream[vid] << sizeWrite;
111  uint32_t rIdx = read_idx[vid];
112  gmem_read:
113  for (uint32_t midx = 0; midx < burstSize; midx++) {
114 #pragma HLS PIPELINE II = 1
115  outStream[vid] << in[rIdx + midx];
116  }
117  read_idx[vid] += burstSize;
118  }
119  }
120  }
121 
122 size_init:
123  for (uint8_t vid = 0; vid < NUM_BLOCKS; vid++) {
124 #pragma HLS UNROLL
125  outLz77SizeStream[vid] << 0;
126  outSizeStream[vid] << 0;
127  }
128 }

201 template <int DATAWIDTH, int BURST_SIZE, int NUM_BLOCKS>
202 void mm2sNb(const ap_uint<DATAWIDTH>* in,
203  const uint32_t _input_idx[NUM_BLOCKS],
204  hls::stream<ap_uint<DATAWIDTH> > outStream[NUM_BLOCKS],
205  const uint32_t _input_size[NUM_BLOCKS]) {
221  const int c_byteSize = 8;
222  const int c_wordSize = DATAWIDTH / c_byteSize;
223  ap_uint<DATAWIDTH> local_buffer[NUM_BLOCKS][BURST_SIZE];
224 #pragma HLS ARRAY_PARTITION variable = local_buffer dim = 1 complete
225 #pragma HLS BIND_STORAGE variable = local_buffer type = RAM_2P impl = LUTRAM
226  uint32_t read_idx[NUM_BLOCKS];
227  uint32_t write_idx[NUM_BLOCKS];
228  uint32_t read_size[NUM_BLOCKS];
229  uint32_t input_idx[NUM_BLOCKS];
230  uint32_t input_size[NUM_BLOCKS];
231 #pragma HLS ARRAY_PARTITION variable = read_idx dim = 0 complete
232 #pragma HLS ARRAY_PARTITION variable = write_idx dim = 0 complete
233 #pragma HLS ARRAY_PARTITION variable = read_size dim = 0 complete
234  ap_uint<NUM_BLOCKS> pending;
235  ap_uint<NUM_BLOCKS> is_full;
236  for (uint32_t bIdx = 0; bIdx < NUM_BLOCKS; bIdx++) {
237 #pragma HLS UNROLL
238  read_idx[bIdx] = 0;
239  write_idx[bIdx] = 0;
240  read_size[bIdx] = 0;
241  input_idx[bIdx] = _input_idx[bIdx];
242  input_size[bIdx] = _input_size[bIdx];
243  pending.range(bIdx, bIdx) = 1;
244  }
245  while (pending) {
246  pending = 0;
247  for (uint32_t bIdx = 0; bIdx < NUM_BLOCKS; bIdx++) {
248  uint32_t pending_bytes = (input_size[bIdx] > read_size[bIdx]) ? (input_size[bIdx] - read_size[bIdx]) : 0;
249  if ((pending_bytes) && (read_idx[bIdx] == write_idx[bIdx])) {
250  uint32_t pending_words = (pending_bytes - 1) / c_wordSize + 1;
251  uint32_t burst_size = (pending_words > BURST_SIZE) ? BURST_SIZE : pending_words;
252  uint32_t mem_read_byte_idx = read_size[bIdx] + input_idx[bIdx];
253  uint32_t mem_read_word_idx = (mem_read_byte_idx) ? ((mem_read_byte_idx - 1) / c_wordSize + 1) : 0;
254  gmem_rd:
255  for (uint32_t i = 0; i < burst_size; i++) {
256 #pragma HLS PIPELINE II = 1
257  local_buffer[bIdx][i] = in[mem_read_word_idx + i];
258  }
259  pending.range(bIdx, bIdx) = 1;
260  read_idx[bIdx] = 0;
261  write_idx[bIdx] = burst_size;
262  read_size[bIdx] += burst_size * c_wordSize;
263  }
264  }
265  ap_uint<NUM_BLOCKS> terminate_all;
266  terminate_all = 1;
267  bool terminate = 0;
268  mm2s:
269  for (int i = 0; (terminate == 0) && (terminate_all != 0); i++) {
270 #pragma HLS PIPELINE II = 1
271  for (uint8_t pb = 0; pb < NUM_BLOCKS; pb++) {
272 #pragma HLS UNROLL
273  is_full.range(pb, pb) = outStream[pb].full();
274  if (!is_full.range(pb, pb) && (read_idx[pb] != write_idx[pb])) {
275  outStream[pb] << local_buffer[pb][read_idx[pb]];
276  read_idx[pb] += 1;
277  }
278  }
279  terminate = 0;
280  for (uint32_t bIdx = 0; bIdx < NUM_BLOCKS; bIdx++) {
281 #pragma HLS UNROLL
282  if (read_idx[bIdx] == write_idx[bIdx]) {
283  terminate_all.range(bIdx, bIdx) = 0;
284  if (read_size[bIdx] < input_size[bIdx]) {
285  terminate = 1;
286  }
287  } else {
288  terminate_all.range(bIdx, bIdx) = 1;
289  pending.range(bIdx, bIdx) = 1;
290  }
291  }
292  }
293  }

638  const int c_byteSize = 8;
639  const int c_wordSize = DATAWIDTH / c_byteSize;
640  ap_uint<DATAWIDTH> local_buffer[NUM_BLOCKS][BURST_SIZE];
641 #pragma HLS ARRAY_PARTITION variable = local_buffer dim = 1 complete
642 #pragma HLS BIND_STORAGE variable = local_buffer type = RAM_2P impl = LUTRAM
643  uint32_t read_idx[NUM_BLOCKS];
644  uint32_t write_idx[NUM_BLOCKS];
645  uint32_t read_size[NUM_BLOCKS];
646  uint32_t input_idx[NUM_BLOCKS];
647  uint32_t input_size[NUM_BLOCKS];
648 #pragma HLS ARRAY_PARTITION variable = read_idx dim = 0 complete
649 #pragma HLS ARRAY_PARTITION variable = write_idx dim = 0 complete
650 #pragma HLS ARRAY_PARTITION variable = read_size dim = 0 complete
651  ap_uint<NUM_BLOCKS> pending;
652  ap_uint<NUM_BLOCKS> is_full;
653  for (uint32_t bIdx = 0; bIdx < NUM_BLOCKS; bIdx++) {
654 #pragma HLS UNROLL
655  read_idx[bIdx] = 0;
656  write_idx[bIdx] = 0;
657  read_size[bIdx] = 0;
658  input_idx[bIdx] = _input_idx[bIdx];
659  input_size[bIdx] = _input_size[bIdx] + (input_idx[bIdx] % c_wordSize);
660  pending.range(bIdx, bIdx) = 1;
661  }
662  while (pending) {
663  pending = 0;
664  for (uint32_t bIdx = 0; bIdx < NUM_BLOCKS; bIdx++) {
665  uint32_t pending_bytes = (input_size[bIdx] > read_size[bIdx]) ? (input_size[bIdx] - read_size[bIdx]) : 0;
666  if ((pending_bytes) && (read_idx[bIdx] == write_idx[bIdx])) {
667  uint32_t pending_words = (pending_bytes - 1) / c_wordSize + 1;
668  uint32_t burst_size = (pending_words > BURST_SIZE) ? BURST_SIZE : pending_words;
669  uint32_t mem_read_byte_idx = read_size[bIdx] + input_idx[bIdx];
670  uint32_t mem_read_word_idx = 0;
671  if (mem_read_byte_idx)
672  mem_read_word_idx = (mem_read_byte_idx % c_wordSize) ? (mem_read_byte_idx - 1) / c_wordSize
673  : ((mem_read_byte_idx - 1) / c_wordSize + 1);
674  else
675  mem_read_word_idx = 0;
676 
677  gmem_rd:
678  for (uint32_t i = 0; i < burst_size; i++) {
679 #pragma HLS PIPELINE II = 1
680  local_buffer[bIdx][i] = in[mem_read_word_idx + i];
681  }
682  pending.range(bIdx, bIdx) = 1;
683  read_idx[bIdx] = 0;
684  write_idx[bIdx] = burst_size;
685  read_size[bIdx] += burst_size * c_wordSize;
686  }
687  }
688  ap_uint<NUM_BLOCKS> terminate_all;
689  terminate_all = 1;
690  bool terminate = 0;
691  mm2s:
692  for (int i = 0; (terminate == 0) && (terminate_all != 0); i++) {
693 #pragma HLS PIPELINE II = 1
694  for (uint8_t pb = 0; pb < NUM_BLOCKS; pb++) {
695 #pragma HLS UNROLL
696  is_full.range(pb, pb) = outStream[pb].full();
697  if (!is_full.range(pb, pb) && (read_idx[pb] != write_idx[pb])) {
698  outStream[pb] << local_buffer[pb][read_idx[pb]];
699  read_idx[pb] += 1;
700  }
701  }
702  terminate = 0;
703  for (uint32_t bIdx = 0; bIdx < NUM_BLOCKS; bIdx++) {
704 #pragma HLS UNROLL
705  if (read_idx[bIdx] == write_idx[bIdx]) {
706  terminate_all.range(bIdx, bIdx) = 0;
707  if (read_size[bIdx] < input_size[bIdx]) {
708  terminate = 1;
709  }
710  } else {
711  terminate_all.range(bIdx, bIdx) = 1;
712  pending.range(bIdx, bIdx) = 1;
713  }
714  }
715  }
716  }
717 }

777  uintMemWidth_t* head_prev_blk,

790 #pragma HLS BIND_STORAGE variable = buffer type = RAM_2P impl = LUTRAM
791 
792  uint32_t offset_gmem = offset ? offset / 64 : 0;
793 
794  // Handle header or residue here
795  uint32_t block_stride = block_size_in_kb * 1024 / 64;
796 
797  uint32_t blkCompSize = 0;
798  uint32_t origSize = 0;
799  uint32_t sizeInWord = 0;
800  uint32_t byteSize = 0;
801  // Run over number of blocks
802  for (int bIdx = 0; bIdx < no_blocks + 1; bIdx++) {
803  if (bIdx == 0) {

805  byteSize = head_res_size;
806  } else {

808  origSize = in_block_size[bIdx - 1];

810  // into streams for next block

812  byteSize = blkCompSize;
813  }

819  // __FUNCTION__,

822  // Copy data from global memory to local
823  // Put it into stream

825  uint32_t chunk_size = BURST_SIZE;
826 
827  if (i + BURST_SIZE > sizeInWord) chunk_size = sizeInWord - i;
828 
829  if (bIdx == 0) {

832 #pragma HLS PIPELINE II = 1

838 #pragma HLS PIPELINE II = 1
